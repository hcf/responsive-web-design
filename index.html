#3 Media queries

http://flic.kr/p/64GVLT

CSS2 media typer 

<link rel="stylesheet" media="screen" href="/screen.css">
<link rel="stylesheet" media="print" href="/print.css">
@media screen {
  body {
    background: #000;
    color: #fff;
  }
}
@media print {
  body {
    background: #fff;
    color: #000;
  }
}

CSS3 media queries 

<link rel="stylesheet" media="screen and (max-width: 35em)"
      href="/narrow.css">
@media screen and (max-width: 35em) {
  #page > article,
  #page > aside {
    float: none;
    width: auto;
  }
  #page > article {
    margin-right: 0;
  }
}

Vertikal utlising av tabell

thead,
tbody,
tr,
th,
td {
  display: block;
}

tr {
  margin-bottom: .5em;
}

Normal tabellstil for bredere skjerm 

@media screen and (min-width: 45em) {
  table {
    display: table;
  }

  tr {
    display: table-row;
  }

  thead {
    display: table-header-group;
  }

  tbody {
    display: table-row-group;
  }

  td,
  th {
    display: table-cell;
  }
}

var responsiveTables = function() {
  $("article table").each(function(table) {
    if($(table).css("display") === "block") {
      prependHeadersToTableCells(table);
    } else if($(table).css("display") === "table") {
      removeHeadersFromTableCells(table);
    }
  });
}
